83   Naive approach ( O(n2) ) is to run two loops to consider all possible pairs. For every pair, check if product is equal to x or not. 
84   // A simple C++ program to find if there is a pair 
85   // with given product. 
86   #include<bits/stdc++.h> 
87   using namespace std; 
88    
89   // Returns true if there is a pair in arr[0..n-1] 
90   // with product equal to x. 
91   bool isProduct(int arr[], int n, int x) 
92   { 
93       // Consider all possible pairs and check for 
94       // every pair. 
95       for (int i=0; i<n-1; i++) 
96          for (int j=i+1; i<n; i++) 
97             if (arr[i] * arr[j] == x) 
98                 return true; 
99    
100      return false; 
101  }
